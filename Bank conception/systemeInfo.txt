import os
import ollama

# Dossier où seront sauvegardés les fichiers de routines
ROUTINES_DIR = "routines"
os.makedirs(ROUTINES_DIR, exist_ok=True)

# Serveur Ollama
OLLAMA_HOST = "http://127.0.0.1:11434"
os.environ["OLLAMA_HOST"] = OLLAMA_HOST

# -----------------------------
# Instructions générales / prompts
# -----------------------------
prompts = {
    "calcul_moyenne": "Écris une fonction Python qui calcule la moyenne d'une liste de nombres, avec gestion des erreurs.",
    "tri_bulle": "Écris une fonction Python pour trier une liste avec l'algorithme du tri à bulles.",
    "fibonacci": "Écris une fonction Python qui renvoie les n premiers nombres de Fibonacci.",
    "lecture_csv": "Écris une fonction Python pour lire un fichier CSV et retourner une liste de dictionnaires.",
    "sauvegarde_json": "Écris une fonction Python pour sauvegarder un dictionnaire dans un fichier JSON."
}

# -----------------------------
# Boucle pour générer les fichiers
# -----------------------------
for filename, prompt in prompts.items():
    # Envoi du prompt à LLaMA
    response = ollama.chat(
        model="llama3:latest",
        messages=[{"role": "user", "content": prompt}]
    )

    code = response['message']['content']

    # Nettoyage rapide pour retirer les ``` si LLaMA les ajoute
    if code.startswith("```") and code.endswith("```"):
        code = "\n".join(code.strip("```").split("\n")[1:])

    # Chemin complet du fichier
    file_path = os.path.join(ROUTINES_DIR, f"{filename}.py")

    # Sauvegarde du code dans le fichier
    with open(file_path, "w", encoding="utf-8") as f:
        f.write(code)

    print(f"[✓] Routine '{filename}' générée et sauvegardée dans {file_path}")

print("\nToutes les routines ont été générées automatiquement ✅")
